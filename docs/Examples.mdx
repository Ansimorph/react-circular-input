import { useState } from 'react'
import { Playground } from 'docz'
import {
	CircularInput,
	CircularProgress,
	CircularThumb,
	CircularTrack,
} from '../src'
import { AnimatedThumb } from './components/AnimatedThumb'

# Examples

## Fully featured

<Playground>
	{() => {
		const [value, setValue] = useState(0.25)
		return (
			<CircularInput value={value} onChange={setValue}>
				<CircularTrack />
				<CircularProgress />
				<CircularThumb />
			</CircularInput>
		)
	}}
</Playground>

## Readonly

Omitting the `onChange` prop makes it readonly.

<Playground>
	{
		<CircularInput value={0.25}>
			<CircularTrack />
			<CircularProgress />
			<CircularThumb />
		</CircularInput>
	}
</Playground>

## Progress

<Playground>
	{
		<CircularInput value={0.25}>
			<CircularProgress />
		</CircularInput>
	}
</Playground>

## Customised

`CircularTrack`, `CircularProgress` and `CircularThumb` are just a SVG `circle`'s so you can tweak most (see Component docs) attributes ðŸ’…

<Playground>
	{() => {
		const [value, setValue] = useState(0.25)
		return (
			<CircularInput value={value} onChange={setValue}>
				<CircularTrack strokeWidth={10} stroke="#ebebeb" />
				<CircularProgress strokeWidth={40} stroke="rgb(255,0,0)" />
				<CircularThumb r={10} fill="rgba(255,255,255,0.5)" />
			</CircularInput>
		)
	}}
</Playground>

## Animated

You can use libraries like `react-spring` to add animation.

<Playground>
	{() => {
		const [value, setValue] = useState(0.25)
		return (
			<CircularInput value={value} onChange={setValue}>
				<CircularTrack />
				<CircularProgress />
				<AnimatedThumb />
			</CircularInput>
		)
	}}
</Playground>

A declarative and composable approach means we have a lot of flexibility, check out the next sections for how to go further!
